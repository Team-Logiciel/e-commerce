/**
 * Trouver une commande par id 
 * @param {*} req 
 * @param {*} res 
 * @param {*} next 
 * @param {*} commandeId 
 */
exports.commandeParId = (req,res,next,commandeId) =>{
    Commande.findById(commandeId)
    .populate('products.product','name price')
    .exec((err,commande)=>{
        if(err){
            res.status(400).json({
                error: errorHandler(err)
            })
        }
        req.commande = commande
        next()
    })

}
